Webpack

Что такое Webpack

  • Module bundler
  • Понимает CommonJS, AMD и ES6 modules
  • Множественные Entrypoints
  • Воспринимает все assets как модули
  • Позволяет трансформировать модули

Множественные Entrypoints

module.exports = { entry: { entry1: './entry1', entry2: './entry2' }, output: { path: 'output', filename: 'bundle-[name].js' }, plugins: [ new CommonsChunkPlugin('common', 'bundle-[name].js') ] };

Loaders

module.exports = { module: { loaders: [ { test: /\.(gif|jpe?g|png)$/, loader: "url?limit=10000" }, { test: /\.css$/, loader: "style!css" }, { test: /\.styl$/, loader: "style!css!stylus" } ] } };

Pre/Post Loaders

module.exports = { module: { preLoaders: [{ test: /\.js$/, exclude: /(node_modules)\//, loader: "eslint" }], postLoaders: [{ test: /\.js$/, exclude: /(test|node_modules)\//, loader: 'istanbul-instrumenter' }] } };

Load On Demand

var a = require("./a"); a.foo(console.log.bind(console)); a.bar(console.log.bind(console)); entry.js module.exports = { foo: function (callback) { callback("foo"); }, bar: function (callback) { require.ensure(["./b"], function (require) { require("./b").bar(callback); }); } }; a.js module.exports = { bar: function (callback) { callback("bar"); } }; b.js
